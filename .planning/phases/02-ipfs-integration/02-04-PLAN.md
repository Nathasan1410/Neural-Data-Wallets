    'use client'

    import { useWriteContract, useReadContract, useWatchContractEvent } from 'wagmi'
    import { useQueryClient } from '@tanstack/react-query'
    import { NEURAL_DATA_CONTRACT, NEURAL_DATA_ABI } from '@/lib/contracts/neuralDataRegistry'
    import { Address } from 'viem'

    export function useHasAccess(userAddress: Address | undefined, researcherAddress: Address | undefined) {
      return useReadContract({
        address: NEURAL_DATA_CONTRACT,
        abi: NEURAL_DATA_ABI,
        functionName: 'hasAccess',
        args: userAddress && researcherAddress ? [userAddress, researcherAddress] : undefined as any,
        query: {
          enabled: !!userAddress && !!researcherAddress
        }
      })
    }

    export function useGrantAccess(onSuccess?: () => void, onError?: (error: string) => void) {
      const queryClient = useQueryClient()

      const { writeContract, isPending, error, reset } = useWriteContract()

      // Invalidate queries when AccessGranted event is emitted
      useWatchContractEvent({
        address: NEURAL_DATA_CONTRACT,
        eventName: 'AccessGranted',
        onLogs: () => {
          queryClient.invalidateQueries({ queryKey: ['hasAccess'] })
        }
      })

      const grantAccess = (researcherAddress: Address) => {
        writeContract(
          {
            address: NEURAL_DATA_CONTRACT,
            abi: NEURAL_DATA_ABI,
            functionName: 'grantAccess',
            args: [researcherAddress]
          },
          {
            onSuccess: () => {
              queryClient.invalidateQueries({ queryKey: ['hasAccess'] })
              onSuccess?.()
            },
            onError: (err) => {
              onError?.(err.message)
            }
          }
        )
      }

      return { grantAccess, isPending, error, reset }
    }

    export function useRevokeAccess(onSuccess?: () => void, onError?: (error: string) => void) {
      const queryClient = useQueryClient()

      const { writeContract, isPending, error, reset } = useWriteContract()

      // Invalidate queries when AccessRevoked event is emitted
      useWatchContractEvent({
        address: NEURAL_DATA_CONTRACT,
        eventName: 'AccessRevoked',
        onLogs: () => {
          queryClient.invalidateQueries({ queryKey: ['hasAccess'] })
        }
      })

      const revokeAccess = (researcherAddress: Address) => {
        writeContract(
          {
            address: NEURAL_DATA_CONTRACT,
            abi: NEURAL_DATA_ABI,
            functionName: 'revokeAccess',
            args: [researcherAddress]
          },
          {
            onSuccess: () => {
              queryClient.invalidateQueries({ queryKey: ['hasAccess'] })
              onSuccess?.()
            },
            onError: (err) => {
              onError?.(err.message)
            }
          }
        )
      }

      return { revokeAccess, isPending, error, reset }
    }
    ```

    'use client'

    import { useState } from 'react'
    import { useGrantAccess } from '@/lib/hooks/useAccessControl'
    import { Address } from 'viem'

    interface GrantAccessButtonProps {
      researcherAddress: string
      onSuccess?: () => void
    }

    export function GrantAccessButton({ researcherAddress, onSuccess }: GrantAccessButtonProps) {
      const [inputValue, setInputValue] = useState(researcherAddress)
      const [error, setError] = useState<string | null>(null)

      const { grantAccess, isPending } = useGrantAccess(
        () => {
          onSuccess?.()
          setError(null)
        },
        (err) => setError(err)
      )

      const handleSubmit = (e: React.FormEvent) => {
        e.preventDefault()

        // Basic address validation
        if (!inputValue?.startsWith('0x') || inputValue.length !== 42) {
          setError('Invalid wallet address format')
          return
        }

        grantAccess(inputValue as Address)
      }

      return (
        <form onSubmit={handleSubmit} className="space-y-2">
          <div className="flex gap-2">
            <input
              type="text"
              value={inputValue}
              onChange={(e) => setInputValue(e.target.value)}
              placeholder="0x..."
              className="flex-1 px-3 py-2 border border-gray-300 rounded text-sm font-mono"
              disabled={isPending}
            />
            <button
              type="submit"
              disabled={isPending}
              className="px-4 py-2 bg-green-600 text-white rounded hover:bg-green-700 disabled:bg-gray-400"
            >
              {isPending ? 'Granting...' : 'Grant Access'}
            </button>
          </div>
          {error && <p className="text-red-600 text-sm">{error}</p>}
        </form>
      )
    }

    'use client'

    import { useState } from 'react'
    import { useRevokeAccess } from '@/lib/hooks/useAccessControl'
    import { Address } from 'viem'

    interface RevokeAccessButtonProps {
      researcherAddress: string
      onSuccess?: () => void
    }

    export function RevokeAccessButton({ researcherAddress, onSuccess }: RevokeAccessButtonProps) {
      const [isConfirming, setIsConfirming] = useState(false)
      const [error, setError] = useState<string | null>(null)

      const { revokeAccess, isPending } = useRevokeAccess(
        () => {
          setIsConfirming(false)
          setError(null)
          onSuccess?.()
        },
        (err) => setError(err)
      )

      const handleRevoke = () => {
        setIsConfirming(true)
        revokeAccess(researcherAddress as Address)
      }

      const isLoading = isPending

      return (
        <div className="space-y-2">
          {!isConfirming ? (
            <button
              onClick={() => setIsConfirming(true)}
              className="px-3 py-1 text-sm text-red-600 border border-red-600 rounded hover:bg-red-50"
            >
              Revoke Access
            </button>
          ) : (
            <div className="flex gap-2 items-center">
              <span className="text-sm text-gray-600">Revoke access to {researcherAddress.slice(0, 10)}...?</span>
              <button
                onClick={handleRevoke}
                disabled={isLoading}
                className="px-3 py-1 text-sm bg-red-600 text-white rounded hover:bg-red-700 disabled:bg-gray-400"
              >
                {isLoading ? 'Revoking...' : 'Confirm'}
              </button>
              <button
                onClick={() => setIsConfirming(false)}
                disabled={isLoading}
                className="px-3 py-1 text-sm text-gray-600 hover:text-gray-800"
              >
                Cancel
              </button>
            </div>
          )}
          {error && <p className="text-red-600 text-sm">{error}</p>}
        </div>
      )
    }
    ```
