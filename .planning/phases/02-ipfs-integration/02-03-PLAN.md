    ```typescript
    'use client'

    import { useWriteContract, useWaitForTransactionReceipt } from 'wagmi'
    import { NEURAL_DATA_CONTRACT, NEURAL_DATA_ABI } from '@/lib/contracts/neuralDataRegistry'
    import { useState } from 'react'
    import { UserRejectedRequestError } from 'viem'

    interface UploadButtonProps {
      cid: string
      onSuccess?: (dataId: bigint) => void
      onError?: (error: string) => void
    }

    export function UploadButton({ cid, onSuccess, onError }: UploadButtonProps) {
      const [step, setStep] = useState<'idle' | 'confirming' | 'success'>('idle')

      const {
        data: hash,
        writeContract,
        isPending,
        error: writeError
      } = useWriteContract()

      const { isLoading: isConfirming, isSuccess: isConfirmed } = useWaitForTransactionReceipt({
        hash
      })

      const handleStoreCid = () => {
        setStep('confirming')
        writeContract(
          {
            address: NEURAL_DATA_CONTRACT,
            abi: NEURAL_DATA_ABI,
            functionName: 'uploadData',
            args: [cid]
          },
          {
            onError: (err) => {
              if (err instanceof UserRejectedRequestError) {
                onError?.('Transaction cancelled')
              } else {
                onError?.(err.message)
              }
              setStep('idle')
            },
            onSuccess: () => {
              setStep('success')
            }
          }
        )
      }

      // Update step based on transaction state
      if (isConfirmed && step === 'confirming') {
        onSuccess?.(0n) // dataId from event logs would be better
        setStep('success')
      }

      const isLoading = isPending || isConfirming
      const isDisabled = isLoading || step === 'success'

      return (
        <button
          onClick={handleStoreCid}
          disabled={isDisabled}
          className="px-4 py-2 bg-blue-600 text-white rounded hover:bg-blue-700 disabled:bg-gray-400 disabled:cursor-not-allowed"
        >
          {isLoading ? 'Storing on-chain...' : step === 'success' ? 'Stored!' : 'Store CID in Contract'}
        </button>
      )
    }
    ```

  <action>
    Create src/components/DataUpload.tsx with complete upload flow:

    ```typescript
    'use client'

    import { useState } from 'react'
    import { useAccount } from 'wagmi'
    import { generateMockEegData, eegDataToJson } from '@/lib/mockEegData'
    import { UploadButton } from './UploadButton'

    type UploadState = 'idle' | 'uploading' | 'uploaded' | 'storing' | 'success' | 'error'

    export function DataUpload() {
      const { address } = useAccount()
      const [state, setState] = useState<UploadState>('idle')
      const [cid, setCid] = useState<string | null>(null)
      const [errorMessage, setErrorMessage] = useState<string | null>(null)

      const handleGenerateAndUpload = async () => {
        if (!address) {
          setErrorMessage('Please connect wallet first')
          return
        }

        setState('uploading')
        setErrorMessage(null)

        try {
          // Generate mock EEG data
          const eegData = generateMockEegData(address, 60, 256)
          const jsonBlob = new Blob([eegDataToJson(eegData)], { type: 'application/json' })

          // Upload to IPFS via API route
          const formData = new FormData()
          formData.append('file', jsonBlob, 'eeg-data.json')

          const response = await fetch('/api/ipfs/upload', {
            method: 'POST',
            body: formData
          })

          if (!response.ok) {
            const error = await response.json()
            throw new Error(error.error || 'Upload failed')
          }

          const result = await response.json()
          setCid(result.cid)
          setState('uploaded')
        } catch (err) {
          setErrorMessage(err instanceof Error ? err.message : 'Upload failed')
          setState('error')
        }
      }

      return (
        <div className="p-6 border border-gray-200 rounded-lg">
          <h2 className="text-xl font-bold mb-4">Upload Neural Data</h2>

          {state === 'idle' && (
            <button
              onClick={handleGenerateAndUpload}
              className="px-4 py-2 bg-green-600 text-white rounded hover:bg-green-700"
            >
              Generate & Upload Mock EEG Data
            </button>
          )}

          {state === 'uploading' && (
            <div className="flex items-center gap-2">
              <div className="animate-spin h-5 w-5 border-2 border-blue-600 border-t-transparent rounded-full" />
              <span>Uploading to IPFS...</span>
            </div>
          )}

          {state === 'uploaded' && cid && (
            <div className="space-y-4">
              <div className="text-sm">
                <p className="text-gray-600">CID:</p>
                <p className="font-mono text-xs break-all">{cid}</p>
              </div>
              <UploadButton
                cid={cid}
                onSuccess={() => setState('success')}
                onError={(err) => {
                  setErrorMessage(err)
                  setState('error')
                }}
              />
            </div>
          )}

          {state === 'success' && (
            <div className="text-green-600 font-medium">
              Data uploaded and CID stored on-chain!
            </div>
          )}

          {state === 'error' && errorMessage && (
            <div className="text-red-600 text-sm">{errorMessage}</div>
          )}
        </div>
      )
    }
    ```
